第1章：C语言入门基础

1.1 什么是C语言

C语言是一种通用的、过程式的编程语言，由Dennis Ritchie在1972年开发。它是二进制工程师和系统程序员的基础语言，因为：

  • 接近底层：C语言提供了直接访问内存的能力
  • 高效性能：编译后的代码执行效率高
  • 广泛应用：操作系统、编译器、嵌入式系统都用C编写
  • 理解基础：学习C语言有助于理解计算机底层原理

1.2 第一个C程序

让我们从经典的“Hello, World!“程序开始：

#include <stdio.h>

int main() {
    printf("Hello, World!\n");
    return 0;
}

在线运行此代码

程序解析

  1. #include <stdio.h>

    • 预处理指令，包含标准输入输出库
    • stdio.h 提供了 printf 等函数

  2. int main()

    • 程序入口点，每个C程序必须有一个main函数
    • int 表示返回整数类型
    • () 表示无参数

  3. printf("Hello, World!\n");

    • 输出字符串到控制台
    • \n 是换行符

  4. return 0;

    • 返回0表示程序正常结束
    • 非0值表示程序异常

编译和运行

# 编译
gcc hello.c -o hello

# 运行
./hello        # Linux/macOS
hello.exe      # Windows

1.3 数据类型

C语言提供了多种基本数据类型，理解这些类型对二进制工程师至关重要：

整数类型

类型大小（字节）范围
char1-128 到 127
short2-32,768 到 32,767
int4-2,147,483,648 到 2,147,483,647
long4或8取决于系统
long long8更大的范围

浮点类型

类型大小（字节）精度
float4单精度
double8双精度
long double10或16扩展精度

示例代码

#include <stdio.h>
#include <limits.h>

int main() {
    // 整数类型
    int age = 25;
    short year = 2024;
    long population = 8000000000L;
    
    // 浮点类型
    float pi = 3.14159f;
    double e = 2.71828;
    
    // 字符类型
    char grade = 'A';
    
    // 打印类型大小
    printf("int大小: %zu 字节\n", sizeof(int));
    printf("float大小: %zu 字节\n", sizeof(float));
    printf("double大小: %zu 字节\n", sizeof(double));
    
    // 打印取值范围
    printf("int最大值: %d\n", INT_MAX);
    printf("int最小值: %d\n", INT_MIN);
    
    return 0;
}

重要提示：sizeof 运算符返回类型或变量占用的字节数，这对理解内存布局非常重要！

1.4 变量和常量

变量声明和初始化

#include <stdio.h>

int main() {
    // 声明变量
    int count;
    float price;
    char symbol;
    
    // 初始化变量
    count = 10;
    price = 99.99f;
    symbol = '$';

    int total = 100;
    double rate = 0.05;
    
    printf("count = %d\n", count);
    printf("price = %.2f\n", price);
    printf("symbol = %c\n", symbol);

常量

#include <stdio.h>

#define MAX_SIZE 100        // 宏定义常量
#define PI 3.14159

int main() {
    const int MIN_AGE = 18;  // const关键字定义常量
    
    printf("最大大小: %d\n", MAX_SIZE);
    printf("圆周率: %.5f\n", PI);
    printf("最小年龄: %d\n", MIN_AGE);

    return 0;

1.5 运算符

算术运算符

#include <stdio.h>

int main() {
    int a = 10, b = 3;
    
    printf("a = %d, b = %d\n", a, b);
    printf("a + b = %d\n", a + b);   // 加法
    printf("a - b = %d\n", a - b);   // 减法
    printf("a * b = %d\n", a * b);   // 乘法
    printf("a / b = %d\n", a / b);   // 整数除法
    printf("a %% b = %d\n", a % b);  // 取余（注意%%转义）
    
    // 浮点除法
    float x = 10.0f, y = 3.0f;
    printf("x / y = %.2f\n", x / y);
    
    return 0;
}

关系运算符

#include <stdio.h>

int main() {
    int a = 10, b = 20;
    
    printf("%d == %d: %d\n", a, b, a == b);  // 等于
    printf("%d != %d: %d\n", a, b, a != b);  // 不等于
    printf("%d < %d: %d\n", a, b, a < b);    // 小于
    printf("%d > %d: %d\n", a, b, a > b);    // 大于
    printf("%d <= %d: %d\n", a, b, a <= b);  // 小于等于
    printf("%d >= %d: %d\n", a, b, a >= b);  // 大于等于
    
    return 0;
}

逻辑运算符

#include <stdio.h>
#include <stdbool.h>

int main() {
    bool x = true, y = false;
    
    printf("x && y: %d\n", x && y);  // 逻辑与
    printf("x || y: %d\n", x || y);  // 逻辑或
    printf("!x: %d\n", !x);          // 逻辑非
    
    return 0;
}

位运算符（重要！）

位运算符对二进制工程师非常重要：

#include <stdio.h>

int main() {
    unsigned int a = 0b10101100;  // 二进制：10101100
    unsigned int b = 0b11110000;  // 二进制：11110000
    
    printf("a = 0x%x (%d)\n", a, a);
    printf("b = 0x%x (%d)\n", b, b);
    
    printf("a & b = 0x%x\n", a & b);  // 按位与
    printf("a | b = 0x%x\n", a | b);  // 按位或
    printf("a ^ b = 0x%x\n", a ^ b);  // 按位异或
    printf("~a = 0x%x\n", ~a);        // 按位取反
    printf("a << 2 = 0x%x\n", a << 2); // 左移
    printf("a >> 2 = 0x%x\n", a >> 2); // 右移
    
    return 0;
}

位运算应用：

  • 设置标志位：flags |= FLAG_BIT
  • 清除标志位：flags &= ~FLAG_BIT
  • 检查标志位：if (flags & FLAG_BIT)
  • 切换标志位：flags ^= FLAG_BIT

1.6 输入输出

printf格式化输出

#include <stdio.h>

int main() {
    int num = 42;
    float f = 3.14159f;
    char ch = 'A';
    char str[] = "Hello";
    
    // 整数
    printf("十进制: %d\n", num);
    printf("八进制: %o\n", num);
    printf("十六进制: %x (小写)\n", num);
    printf("十六进制: %X (大写)\n", num);
    
    // 浮点数
    printf("浮点数: %f\n", f);
    printf("科学计数: %e\n", f);
    printf("保留2位小数: %.2f\n", f);
    
    // 字符和字符串
    printf("字符: %c\n", ch);
    printf("字符串: %s\n", str);
    
    // 宽度和对齐
    printf("右对齐: %10d\n", num);
    printf("左对齐: %-10d\n", num);
    printf("前导零: %05d\n", num);
    
    return 0;
}

scanf输入

#include <stdio.h>

int main() {
    int age;
    float height;
    char name[50];
    
    printf("请输入年龄: ");
    scanf("%d", &age);  // 注意&符号！
    
    printf("请输入身高: ");
    scanf("%f", &height);
    
    printf("请输入姓名: ");
    scanf("%s", name);  // 字符串不需要&
    
    printf("\n你的信息:\n");
    printf("姓名: %s\n", name);
    printf("年龄: %d\n", age);
    printf("身高: %.2f\n", height);
    
    return 0;
}

重要：scanf 使用 & 获取变量地址，这是指针的基础！

1.7 内存地址初探

理解内存地址是二进制工程师的核心技能：

#include <stdio.h>

int main() {
    int a = 10;
    int b = 20;
    
    printf("变量a的值: %d\n", a);
    printf("变量a的地址: %p\n", (void*)&a);
    printf("变量b的值: %d\n", b);
    printf("变量b的地址: %p\n", (void*)&b);
    
    printf("\n变量大小: %zu 字节\n", sizeof(a));
    printf("地址差: %ld 字节\n", (long)((char*)&b - (char*)&a));
    
    return 0;
}

观察：

  • 每个变量都有唯一的内存地址
  • 地址通常用十六进制表示
  • 相邻变量的地址可能连续（取决于编译器）
