第1章：C语言入门基础

1.1 什么是C语言

C语言是一种通用的、过程式的编程语言，由Dennis Ritchie在1972年开发。它是二进制工程师和系统程序员的基础语言，因为：

  • 接近底层：C语言提供了直接访问内存的能力
  • 高效性能：编译后的代码执行效率高
  • 广泛应用：操作系统、编译器、嵌入式系统都用C编写
  • 理解基础：学习C语言有助于理解计算机底层原理

1.2 第一个C程序

让我们从经典的“Hello, World!“程序开始：

#include <stdio.h>

int main() {
    printf("Hello, World!\n");
    return 0;
}

在线运行此代码

程序解析

  1. #include <stdio.h>

    • 预处理指令，包含标准输入输出库
    • stdio.h 提供了 printf 等函数

  2. int main()

    • 程序入口点，每个C程序必须有一个main函数
    • int 表示返回整数类型
    • () 表示无参数

  3. printf("Hello, World!\n");

    • 输出字符串到控制台
    • \n 是换行符

  4. return 0;

    • 返回0表示程序正常结束
    • 非0值表示程序异常

编译和运行

# 编译
gcc hello.c -o hello

# 运行
./hello        # Linux/macOS
hello.exe      # Windows

1.5 运算符

算术运算符

#include <stdio.h>

int main() {
    int a = 10, b = 3;
    
    printf("a = %d, b = %d\n", a, b);
    printf("a + b = %d\n", a + b);   // 加法
    printf("a - b = %d\n", a - b);   // 减法
    printf("a * b = %d\n", a * b);   // 乘法
    printf("a / b = %d\n", a / b);   // 整数除法
    printf("a %% b = %d\n", a % b);  // 取余（注意%%转义）
    
    // 浮点除法
    float x = 10.0f, y = 3.0f;
    printf("x / y = %.2f\n", x / y);
    
    return 0;
}

关系运算符

#include <stdio.h>

int main() {
    int a = 10, b = 20;
    
    printf("%d == %d: %d\n", a, b, a == b);  // 等于
    printf("%d != %d: %d\n", a, b, a != b);  // 不等于
    printf("%d < %d: %d\n", a, b, a < b);    // 小于
    printf("%d > %d: %d\n", a, b, a > b);    // 大于
    printf("%d <= %d: %d\n", a, b, a <= b);  // 小于等于
    printf("%d >= %d: %d\n", a, b, a >= b);  // 大于等于
    
    return 0;
}

位运算符（重要！）

位运算符对二进制工程师非常重要：

#include <stdio.h>

int main() {
    unsigned int a = 0b10101100;  // 二进制：10101100
    unsigned int b = 0b11110000;  // 二进制：11110000
    
    printf("a = 0x%x (%d)\n", a, a);
    printf("b = 0x%x (%d)\n", b, b);
    
    printf("a & b = 0x%x\n", a & b);  // 按位与
    printf("a | b = 0x%x\n", a | b);  // 按位或
    printf("a ^ b = 0x%x\n", a ^ b);  // 按位异或
    printf("~a = 0x%x\n", ~a);        // 按位取反
    printf("a << 2 = 0x%x\n", a << 2); // 左移
    printf("a >> 2 = 0x%x\n", a >> 2); // 右移
    
    return 0;
}

位运算应用：

  • 设置标志位：flags |= FLAG_BIT
  • 清除标志位：flags &= ~FLAG_BIT
  • 检查标志位：if (flags & FLAG_BIT)
  • 切换标志位：flags ^= FLAG_BIT

scanf输入

#include <stdio.h>

int main() {
    int age;
    float height;
    char name[50];
    
    printf("请输入年龄: ");
    scanf("%d", &age);  // 注意&符号！
    
    printf("请输入身高: ");
    scanf("%f", &height);
    
    printf("请输入姓名: ");
    scanf("%s", name);  // 字符串不需要&
    
    printf("\n你的信息:\n");
    printf("姓名: %s\n", name);
    printf("年龄: %d\n", age);
    printf("身高: %.2f\n", height);
    
    return 0;
}

重要：scanf 使用 & 获取变量地址，这是指针的基础！

1.7 内存地址初探

理解内存地址是二进制工程师的核心技能：

#include <stdio.h>

int main() {
    int a = 10;
    int b = 20;
    
    printf("变量a的值: %d\n", a);
    printf("变量a的地址: %p\n", (void*)&a);
    printf("变量b的值: %d\n", b);
    printf("变量b的地址: %p\n", (void*)&b);
    
    printf("\n变量大小: %zu 字节\n", sizeof(a));
    printf("地址差: %ld 字节\n", (long)((char*)&b - (char*)&a));
    
    return 0;
}

观察：

  • 每个变量都有唯一的内存地址
  • 地址通常用十六进制表示
  • 相邻变量的地址可能连续（取决于编译器）

  • 实现二进制打印函数（可选，也可以手动计算）
  • 每个操作后输出当前状态

  • 检查标志：if (flags & FLAG_READ)
  • 清除标志：flags &= ~FLAG_WRITE
  • 切换标志：flags ^= FLAG_EXECUTE
  • 二进制打印：使用位运算逐位检查

参考答案

#include <stdio.h>

#define FLAG_READ    0x01  // 00000001
#define FLAG_WRITE   0x02  // 00000010
#define FLAG_EXECUTE 0x04  // 00000100

void print_binary(unsigned int n) {
    for (int i = 7; i >= 0; i--) {
        printf("%d", (n >> i) & 1);
    }
}

int main() {
    unsigned int flags = 0;
    
    printf("初始flags: 0x%02x (二进制: ", flags);
    print_binary(flags);
    printf(")\n");
    
    // 设置READ标志
    flags |= FLAG_READ;
    printf("设置READ后: 0x%02x (二进制: ", flags);
    print_binary(flags);
    printf(")\n");
    
    // 设置WRITE标志
    flags |= FLAG_WRITE;
    printf("设置WRITE后: 0x%02x (二进制: ", flags);
    print_binary(flags);
    printf(")\n");
    
    // 检查READ权限
    if (flags & FLAG_READ) {
        printf("有READ权限\n");
    }
    
    // 清除WRITE标志
    flags &= ~FLAG_WRITE;
    printf("清除WRITE后: 0x%02x (二进制: ", flags);
    print_binary(flags);
    printf(")\n");
    
    // 切换EXECUTE标志
    flags ^= FLAG_EXECUTE;
    printf("切换EXECUTE后: 0x%02x (二进制: ", flags);
    print_binary(flags);
    printf(")\n");
    
    return 0;
}

参考答案

#include <stdio.h>

#define MAX_NAME_LEN 50

void print_binary(int n) {
    if (n == 0) {
        printf("0");
        return;
    }
    int bits[32];
    int count = 0;
    while (n > 0) {
        bits[count++] = n % 2;
        n /= 2;
    }
    for (int i = count - 1; i >= 0; i--) {
        printf("%d", bits[i]);
    }
}

int main() {
    char name[MAX_NAME_LEN];
    int age;
    float score;
    
    printf("请输入学生姓名: ");
    scanf("%s", name);
    
    printf("请输入年龄: ");
    scanf("%d", &age);
    
    printf("请输入成绩: ");
    scanf("%f", &score);
    
    printf("\n=== 学生信息 ===\n");
    printf("姓名: %-20s\n", name);
    printf("年龄: %5d\n", age);
    printf("成绩: %.2f\n", score);
    
    // 成绩的十六进制（转换为整数）
    int score_int = (int)score;
    printf("成绩(十六进制): 0x%x\n", score_int);
    
    // 年龄的二进制
    printf("年龄(二进制): ");
    print_binary(age);
    printf("\n");
    
    return 0;
}
